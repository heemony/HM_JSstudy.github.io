'bcaebe' 문자열을 hasOwnProperty 를 사용해서, {a : 1, b : 2, c : 1, e : 2} 로 변경해보세요.


function  getFrequencies(str) {
    var result = {};

    for (var index = 0; index < str.length; index = index + 1) {
        var character = str[index];

        if (result.hasOwnProperty(character)) {
            result[character] = result[character] + 1;
        } else {
            result[character] = 1;
        }
    }
    return result;
}
