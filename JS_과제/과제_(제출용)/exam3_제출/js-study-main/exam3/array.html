[8번 문제]
	[처음에 했던 오류나던 코드]
	examObj8.find('button').on('click', function() {
		var ex8 = [];

		for(let i = 0; i < 6; i++) {
			var n = Math.floor(Math.random() * 5);
			if( ex8.indexOf(n) === -1) { // ex8에서 뽑은 난수 n에 해당하는 숫자가 없다면 
				ex8.push(n); // ex8 배열에 해당 숫자를 넣어주세요.
			} else { // 있다면
				i--; // 증가된 인덱스를 -1 해주고 for문을 나온다.
			}
		};
		examObj8.append(`${ex8}`);	
	});

	[오류가 났던 이유]
	1. 난수 범위 설정 : [5] -> 6, 함수 제대로 이해할 것..ㅠㅠ..
	2. [indexOf] -> includes : 요 차이 정리할 것.
	3. for문 변수 타입 : [var] -> let : 뭐 포문 쓸 땐 var보단 let이 낫다함.



	# 문자열 포함여부는 2가지가 있다.
	- indexOf
	- includes

	둘의 가장 큰 차이는 웹표준때문에 includes는 크롬에서 먹히는데, 익스플로러에선 안된다고 한다
	indexOf는 대부분에서 가능하다고 함.
