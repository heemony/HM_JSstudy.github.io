Q. for/while 문 2개로 구구단을 출력하세요.



// exam4 [s]
var exam4 = $('.exam_q').eq(3);

exam4.append('----for문----')
for( var i = 2; i < 10 ; i++) {
	for ( var j = 1; j < 10 ; j++) {
		exam4.append(`${i} X ${j} = ${i*j} `);
	}
}

exam4.append('----while문----')
var a = 2
var b = 1;
while (a < 10){
	while ( b < 10 ) {
		exam4.append(`${a} X ${b} = ${a*b} `);
		b++;
	}
	b = 1;
	a++;
}
